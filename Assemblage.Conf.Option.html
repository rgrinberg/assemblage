module Option: sig .. end

Options


val wrap : 'a Assemblage.Conf.value option -> 'a option Assemblage.Conf.value
wrap o is o as an optional value.
val some : 'a Assemblage.Conf.value -> 'a option Assemblage.Conf.value
some v is the value of v as a Some value.
val get : ?none:'a Assemblage.Conf.value ->
       'a option Assemblage.Conf.value -> 'a Assemblage.Conf.value
get none v is the value of the option v.
Raises Invalid_argument if the value of v is None and there none is unspecified.
