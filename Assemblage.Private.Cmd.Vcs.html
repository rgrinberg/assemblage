include Assemblage.Cmd.Vcs
val override_kind : unit -> t option
override_kind () is the current VCS kind override value. See Assemblage.Private.Cmd.Vcs.set_override_kind.
val set_override_kind : t option -> unit
set_override_kind (Some vcs) has the effect of bypassing VCS discovery. vcs will always Assemblage.Cmd.Vcs.exists, Assemblage.Cmd.Vcs.find and Assemblage.Cmd.Vcs.get will always always return this vcs.
val override_exec : unit -> string option
override_bin is the current VCS executable override. See Assemblage.Private.Cmd.Vcs.set_override_exec.
val set_override_exec : string option -> unit
set_override_exec (Some vcs_exec) uses vcs_exec as the VCS executable, use in conjunction with Assemblage.Private.Cmd.Vcs.set_override_kind otherwise it has no effect.
